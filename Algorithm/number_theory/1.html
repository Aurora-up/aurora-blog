                    pub fn divisors_sum(mut n: usize) -> usize {
  let mut mp: HashMap<usize, usize> = HashMap::new();
  let sqrt_n = (n as f32).sqrt() as usize;
  for i in 2..sqrt_n + 1 {
    if n % i == 0 {
      let mut cnt = 0;
      while n % i == 0 {
        n /= i;
        cnt += 1;
      }
      mp.insert(i, cnt);
    }
  }
  if n > 1 {
    mp.insert(n, 1);
  }

  let mut res: usize = 1;
  for (k, mut v) in mp {
    let mut t: usize = 1;

pub fn two_sum(nums: Vec<i32>, target: i32) -> Vec<i32> {
  let mut ans: Vec<i32> = vec![];

  //  foo è®°å½•æ¯ä¸ªæ•°çš„åŸç´¢å¼•å’Œ target - nums[index]
  let mut foo: Vec<Pair<i32, i32>> = Vec::new();

    foo.push(Pair {
      x: index as i32,
      y: target - *val,
    });
  }

  //  å‡åºæ’åˆ—
  foo.sort_by(|a, b| a.y.cmp(&b.y));

  // å›ºå®šä¸€ä¸ªæ•°ï¼Œç„¶åäºŒåˆ†æœç´¢å¦ä¸€ä¸ªæ•°
  for i in foo.iter() {

    let second_index = foo.binary_search_by_key(&(target - first), |pair| pair.y);
    match second_index {
      Ok(index) => {
        if i.x != foo[index].x {
          ans.push(i.x);
          ans.push(foo[index].x);
          break;
        }
      }
      Err(e) => {
        continue;
      }
    }
  }
  ans.sort();
  ans
}
